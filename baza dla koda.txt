<script type="text/javascript">
  function pole_voloshchuk(a, b, h) {
    var a_vol = parseFloat(a);
    var b_vol = parseFloat(b);
    var h_vol = parseFloat(h);

    var d = 0.5 * (a_vol + b_vol) * h_vol;
    document.getElementsByName('Pole')[0].value = d.toFixed(2);
  }

  function obwod_voloshchuk(a, b, h) {
    var a_vol = parseFloat(a);
    var b_vol = parseFloat(b);
    var h_vol = parseFloat(h);

    var c = Math.abs(a_vol - b_vol) / 2;
    var d = Math.sqrt(Math.pow(c, 2) + Math.pow(h_vol, 2));
    var o = a_vol + b_vol + 2 * d;

    document.getElementsByName('obw')[0].value = o.toFixed(2);
  }
</script>
